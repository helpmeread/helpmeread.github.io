int ary[10][10], completed[10], i, j, n, cost = 0;

void mincost(int city) {
    int ncity;
    completed[city] = 1;

    printf("%d--->", city + 1);
    ncity = least(city);

    if (ncity == 999) {
        ncity = 0;
        printf("%d", ncity + 1);
        cost += ary[city][ncity];

        return;
    }

    mincost(ncity);
}

int least(int c) {
    int nc = 999, min = 999, kmin;

    for (i = 0; i < n; i++) {
        if ((ary[c][i] != 0) && (completed[i] == 0))
            if (ary[c][i] + ary[i][c] < min) {
                min = ary[c][i] + ary[i][0];
                kmin = ary[c][i];
                nc = i;
            }
    } 

    if (min != 999)
        cost += kmin;

    return nc;
}

#include <stdio.h>
#define infinity 999
void dijkstra(int n, int v, int cost[10][10], int dist[100])
{
    int i, u, count, w, flag[10], min;
    for (i = 1; i <= n; i++){
        flag[i] = 0, dist[i] = cost[v][i];
    }
    count = 2;
    while (count <= n)
    {
        min = 99;
        for (w = 1; w <= n; w++){
            if (dist[w] < min && !flag[w])
                min = dist[w], u = w;
        }
        flag[u] = 1;
        count++;
        for (w = 1; w <= n; w++){
            if ((dist[u] + cost[u][w] < dist[w]) && !flag[w])
                dist[w] = dist[u] + cost[u][w];
        }
    }
}

{
    int n, v, i, j, cost[10][10], dist[10];
    printf("\n Enter the number of nodes:");

            if (cost[i][j] == 0)
                cost[i][j] = infinity;
        }
    }

    for (i = 1; i <= n; i++){ 
        if (i != v) {
            printf("%d->%d,cost=%d\n", v, i, dist[i]);
        }
    }
}
        
    

 

#include <stdio.h>
 
int max(int a, int b) { 
    return (a > b)? a : b; 
}
int knapsack(int W, int wt[], int val[], int n)
{
   int i, w;
   int K[n+1][W+1];
 
   for (i = 0; i <= n; i++)
   {
       for (w = 0; w <= W; w++)
       {
           if (i==0 || w==0)
               K[i][w] = 0;
           else if (wt[i-1] <= w)
                 K[i][w] = max(val[i-1] + K[i-1][w-wt[i-1]],  K[i-1][w]);
           else
                 K[i][w] = K[i-1][w];
       }
   }
 
   return K[n][W];
}

#include <stdio.h>
#include <math.h>

int a[30];

int place(int pos) {
    for (int i = 1; i < pos; i++) {
        if (a[i] == a[pos] || abs(a[i] - a[pos]) == abs(i - pos)) {
            return 0;
        }
    }
    return 1;
}

void print_sol(int n) {
    int count=0;
    count++;
    printf("\n\nSolution #%d:\n", count);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            printf(a[i] == j ? "Q\t" : "*\t");
        }
        printf("\n");
    }
}

void queen(int n, int k) {
    if (k > n) {
        print_sol(n);
        return;
    }
    
    for (a[k] = 1; a[k] <= n; a[k]++) {
        if (place(k)) {
            queen(n, k + 1);
        }
    }
}

int main() {
    int n;
    printf("Enter the number of Queens:\n");
    scanf("%d", &n);
    queen(n, 1);
    return 0;
}

    

 
#include <stdio.h>
#define V 5

int graph[V][V], path[V];

int isSafe(int v, int pos, int path[V]) {
    // Check if it's safe to add vertex v to the path at position pos
    return graph[path[pos - 1]][v] && !path[pos];
}

int hamiltonianCycleUtil(int pos) {
    if (pos == V)
        return graph[path[pos - 1]][path[0]]; // Check if path is cyclic

    for (int v = 1; v < V; v++) {
        if (isSafe(v, pos, path)) {
            path[pos] = v; // Add vertex to the path
            if (hamiltonianCycleUtil(pos + 1))
                return 1; // Continue exploring
        }
    }

    return path[pos] = -1; // Backtrack
}

int main() {
    for (int i = 0; i < V; i++)
        path[i] = -1, path[0] = 0;

    printf("Enter the adjacency matrix:\n");
    for (int i = 0; i < V; i++)
        for (int j = 0; j < V; scanf("%d", &graph[i][j++]));

    if (hamiltonianCycleUtil(1)) {
        printf("Hamiltonian Cycle exists:\n");
        for (int i = 0; i < V; printf("%d ", path[i++]));
        printf("%d\n", path[0]);
    } else {
        printf("No Hamiltonian Cycle exists.\n");
    }

    return 0;
}

#define V 4

int isSafe(int graph[V][V], int color[], int vertex, int c)
{
    for (int i = 0; i < V; i++) {
        if (graph[vertex][i] && color[i] == c) {
            return 0;
        }
    }
    return 1;
}

int graphColoring(int graph[V][V], int m, int color[], int vertex)
{
    if (vertex == V) {
        for (int i = 0; i < V; i++) {
            printf(" %d ", color[i]);
        }
        printf("\n");
    }

    for (int c = 1; c <= m; c++) {
        if (isSafe(graph, color, vertex, c)) {
            color[vertex] = c;
            if (graphColoring(graph, m, color, vertex + 1)) {
                return 1;
            }
            color[vertex] = 0;
        }
    }
    return 0;
}
